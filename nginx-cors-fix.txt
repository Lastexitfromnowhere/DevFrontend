server {
    listen 80;
    server_name lastexitvpn.duckdns.org;

    # Redirection vers HTTPS
    location / {
        return 301 https://$host$request_uri;
    }
}

server {
    listen 443 ssl;
    server_name lastexitvpn.duckdns.org;

    # Configuration SSL
    ssl_certificate /etc/letsencrypt/live/lastexitvpn.duckdns.org/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/lastexitvpn.duckdns.org/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_prefer_server_ciphers off;

    # Gestion globale des requêtes OPTIONS (pre-flight)
    if ($request_method = 'OPTIONS') {
        add_header 'Access-Control-Allow-Origin' 'https://wind-frontend-rosy.vercel.app' always;
        add_header 'Access-Control-Allow-Methods' 'GET, POST, PUT, DELETE, OPTIONS' always;
        add_header 'Access-Control-Allow-Headers' 'Content-Type, Authorization, X-Wallet-Address' always;
        add_header 'Access-Control-Max-Age' '86400' always;
        add_header 'Content-Type' 'text/plain charset=UTF-8' always;
        add_header 'Content-Length' '0' always;
        return 204;
    }

    # Redirection vers Vercel pour le frontend
    location / {
        return 301 https://wind-frontend-rosy.vercel.app$request_uri;
    }

    # Configuration pour l'API backend
    location /api {
        proxy_pass http://localhost:10000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        
        # Ne PAS ajouter les en-têtes CORS ici, laisser le backend les gérer
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Augmenter le timeout
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # Configuration pour le service DHT
    location /dht {
        proxy_pass http://localhost:10001/dht;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        
        # Ne PAS ajouter les en-têtes CORS ici, laisser le backend les gérer
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Augmenter le timeout
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # Endpoint spécifique pour le statut
    location = /status {
        # Rediriger vers /dht/status pour la compatibilité
        return 301 https://$host/dht/status;
    }
}
